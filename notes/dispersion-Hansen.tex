\section{Dispersion relation for water free-surface waves with surfactant, following \texorpdfstring{\cite{Hansen1964}}{Hansen \& Mann (1964)}}

\subsection{Navier--Stokes}

Assume an incompressible water body. The derivation works with the velocity potential and the stream function, $v=\nabla\vPotential+\nabla\times\vStream$. For an irrotational fluid, of course, $\vStream=0$. Let $\sDensity$ be the water density, assumed constant, $p$ be the pressure, $g$ be the gravitational acceleration, and $\kinematicViscousity$ be the kinematic viscosity. Navier--Stokes equations then take the form
\begin{align*}
\partial_{t}v
  &= -\nabla\left(\frac{1}{\sDensity}p+gz\right)+\kinematicViscousity\nabla^{2}v \\
\implies\nabla\partial_{t}\vPotential+\nabla\times\partial_{t}\vStream
  &= -\nabla\left(\frac{1}{\sDensity}p+gz\right)+\kinematicViscousity\nabla\times\left(\nabla^{2}\vStream\right)
\end{align*}
where we used the incompressibility, or Laplace, condition $\nabla^{2}\vPotential=0$.

We require that the two components of the velocity field take care separately of each physical effect:
\begin{align*}
\nabla\partial_{t}\vPotential       &= -\nabla\left(\frac{1}{\sDensity}p+gz\right), \\
\nabla\times\partial_{t}\vStream    &= \kinematicViscousity\nabla\times\left(\nabla^{2}\vStream\right).
\end{align*}
Let $\sPressure$ be the pressure at the surface and let $\delta p=p-\sPressure$ be the deviation. Then by a gauge choice, it is clearly enough to demand that
\begin{align}
\delta p                                            &= -\sDensity\partial_{t}\vPotential-\sDensity gz, \label{eq:navier-stokes} \\
\frac{1}{\kinematicViscousity}\partial_{t}\vStream  &= \nabla^{2}\vStream. \nonumber
\end{align}

\subsection{Concentration modeling}

Let $\sBulkCo$ be the pollutant bulk concentration, and let $\sSurfCo$ be the corresponding surfactant concentration. We have two conservation of matter equations and one boundary condition:
\begin{align*}
\partial_{t}\sBulkCo &= D\nabla^{2}\sBulkCo-\nabla\cdot\left(\sBulkCo v\right), \\
\partial_{t}\sSurfCo &= -\partial_{x}\!\left(\sSurfCo v_{x}\!\left(t,x,y,0\right)\right)
                       -\partial_{y}\!\left(\sSurfCo v_{y}\!\left(t,x,y,0\right)\right)+r, \\
\sBulkCo_{0}         &= \sBulkCo\!\left(x,y,z=-\aHeight\right),
\end{align*}
where $r$ is a source term that measures the flow from the bulk into the surface, and $\sBulkCo_{0}$ is the mean bulk concentration.

We assume that both the deviations from time averages, and the fluid velocities, are small, therefore amenable to linearization. Let $\sSurfCo_{0}$ be the mean surfactant concentration. We further suppose that $\sSurfCo\!\left(t,x,y\right)\propto\sBulkCo\!\left(t,x,y,0\right)$, so that $\delta\sSurfCo=\kappa\,\delta\sBulkCo$ where $\kappa=\partial_{\sBulkCo_{0}}\sSurfCo_{0}$ is the first order proportion constant (next order introduces variations). Note that $\kappa$ measures solubility --- a large value means that a large change of concentration at the surface affects but little the concentration in the bulk.

We also assume a rather homogeneous, dilute, chemically and electrically quiet, and strictly static macroscopically, bulk, so the flux between bulk and surface is diffusion limited: $r=-D\partial_{z}\sBulkCo$, $D$ being the diffusion constant.

Therefore
\begin{align*}
\partial_{t}\delta\sBulkCo
  &= D\!\left(\partial_{x}^{2}\delta\sBulkCo+\partial_{y}^{2}\delta\sBulkCo+\partial_{z}^{2}\delta\sBulkCo\right), \\
\kappa\partial_{t}\delta\sBulkCo\!\left(t,x,y,0\right)
  &= -\sSurfCo_{0}\partial_{x}\!\left(\partial_{x}\vPotential\!\left(t,x,y,0\right)-\partial_{z}\vStream_{y}\!\left(t,x,y,0\right)+\partial_{y}\vStream_{z}\!\left(t,x,y,0\right)\right) \\
  &\phantom{=}\,\,-\sSurfCo_{0}\partial_{y}\!\left(\partial_{y}\vPotential\!\left(t,x,y,0\right)+\partial_{z}\vStream_{x}\!\left(t,x,y,0\right)-\partial_{x}\vStream_{z}\!\left(t,x,y,0\right)\right) \\
  &\phantom{=}\,\,-D\partial_{z}\delta\sBulkCo\!\left(t,x,y,0\right), \\
0 &= \delta\sBulkCo\!\left(x,y,z=-\aHeight\right).
\end{align*}
Note that the linear convection term $\sBulkCo_{0}\!\left(\partial_{x}v_{x}\!\left(t,x,y,z\right)+\partial_{z}v_{z}\!\left(t,x,y,z\right)\right)$ is dropped thanks to incompressibility.

\subsection{Surface stress}

For an incompressible fluid, the stress tensor takes the form
\begin{align}
\stressTensor_{ij}
  &= -p\delta_{ij}+\sDensity\kinematicViscousity\!\left(\partial_{i}v_{j}+\partial_{j}v_{i}\right) \nonumber \\
  &= -p\delta_{ij}+\sDensity\kinematicViscousity\!\left(2\partial_{i}\partial_{j}\vPotential+\epsilon_{jkl}\partial_{i}\partial_{k}\vStream_{l}+\epsilon_{ikl}\partial_{j}\partial_{k}\vStream_{l}\right) \label{eq:fluid-stress} \\
\implies\stressTensor_{xz}
  &= \sDensity\kinematicViscousity\!\left(2\partial_{x}\partial_{z}\vPotential+\left(\partial_{x}^{2}-\partial_{z}^{2}\right)\vStream_{y}-\partial_{x}\partial_{y}\vStream_{x}+\partial_{z}\partial_{y}\vStream_{z}\right), \nonumber \\
\stressTensor_{yz}
  &= \sDensity\kinematicViscousity\!\left(2\partial_{y}\partial_{z}\vPotential-\left(\partial_{y}^{2}-\partial_{z}^{2}\right)\vStream_{x}+\partial_{x}\partial_{y}\vStream_{y}-\partial_{x}\partial_{z}\vStream_{z}\right), \nonumber \\
\stressTensor_{zz}
  &= -p+2\sDensity\kinematicViscousity\!\left(\partial_{z}^{2}\vPotential+\partial_{x}\partial_{z}\vStream_{y}-\partial_{y}\partial_{z}\vStream_{x}\right). \nonumber
\end{align}

Gemini claims that for typical sea surfactants, concentration equilibration takes seconds to minutes. In the following we neglect seismic activity in these time scales, which are in general many orders of magnitude weaker than the higher frequency signals that we measured, say at 5\,Hz and above. In this ripple regime, flux between the bulk and the surface is negligible w.r.t.\ the elastic stretching of the surface film in its effect on the instantaneous concentration of the surfactant. Then we get the conservation of matter condition:
\begin{align*}
\sSurfCo\, dA &= dM = \text{const} \\
\implies \sSurfCo_{0}\, dA_{0}
  &= \left(\sSurfCo_{0}+\delta\sSurfCo\right)\left(dA_{0}+\delta dA\right) \\
\implies \frac{\delta\sSurfCo}{\sSurfCo_{0}}
  &= -\frac{\delta dA}{dA_{0}}=\kappa\frac{\delta\sBulkCo}{\sSurfCo_{0}}, \\
\frac{\partial_{t}\delta\sSurfCo}{\sSurfCo_{0}}
  &= -\frac{\partial_{t}dA}{dA_{0}}=-\kappa\frac{\partial_{t}\delta\sBulkCo}{\sSurfCo_{0}}.
\end{align*}
The surface tension per unit length at a point $\left(x,y\right)$ is then to linear order
\begin{align*}
\sTension &= \sTension_{0}+k_{e}\delta dA/dA_{0}+k_{v}\frac{\partial_{t}dA}{dA_{0}} \\
          &= \sTension_{0}-k_{e}\kappa\frac{\delta\sBulkCo}{\sSurfCo_{0}}-k_{v}\kappa\frac{\partial_{t}\delta\sBulkCo}{\sSurfCo_{0}},
\end{align*}
where $k_{e}$ is the film's dilatational elastic constant and $k_{v}$ is the relaxation constant (dilatational viscosity). Now if the air above is still (no wind), the total force acting tangentially on a unit surfactant area per unit length is
\begin{align*}
\frac{dF_{x}}{dx} &= -\stressTensor_{xz}\,dy+\partial_{x}\sTension\,dy, \\
\frac{dF_{y}}{dy} &= -\stressTensor_{yz}\,dx+\partial_{y}\sTension\,dx \\
\implies \frac{dF}{dA}
  &= \begin{pmatrix}-\stressTensor_{xz}+\partial_{x}\sTension\\
                    -\stressTensor_{yz}+\partial_{y}\sTension
     \end{pmatrix}.
\end{align*}
This would accelerate the small area of surfactant. But at the limit of a thin film, the (3D!) mass of this surfactant is minute. Therefore at normal excitation frequencies (finite accelerations), $\frac{dF}{dA}$ is negligible, and
\begin{align}
\stressTensor_{xz}
  &= \partial_{x}\sTension \nonumber \\
  &= -k_{e}\kappa\sSurfCo_{0}^{-1}\partial_{x}\delta\sBulkCo-k_{v}\kappa\sSurfCo_{0}^{-1}\partial_{x}\partial_{t}\delta\sBulkCo, \label{eq:shear-at-boundary} \\
\stressTensor_{yz}
  &= \partial_{y}\sTension \nonumber \\
  &= -k_{e}\kappa\sSurfCo_{0}^{-1}\partial_{y}\delta\sBulkCo-k_{v}\kappa\sSurfCo_{0}^{-1}\partial_{y}\partial_{t}\delta\sBulkCo. \nonumber
\end{align}
Similarly, the normal stress at the surface balances the local pull of the water sheet. For linear ripples the local slopes are small and we approximate
\begin{equation}
\stressTensor_{zz} = -\sPressure-\sDensity g\zeta+\sTension_{0}\partial_{x}^{2}\zeta+\sTension_{0}\partial_{y}^{2}\zeta. \label{eq:pressure-at-boundary}
\end{equation}

\subsection{Recap}

To sum up, we have now a bunch of equations we should mingle together:
\begin{align}
0
  &= \partial_{x}^{2}\vPotential+\partial_{y}^{2}\vPotential+\partial_{z}^{2}\vPotential, \label{eq:modified-Hansens-model} \\
\frac{1}{\kinematicViscousity}\partial_{t}\vStream
  &= \partial_{x}^{2}\vStream+\partial_{y}^{2}\vStream+\partial_{z}^{2}\vStream, \nonumber \\
\frac{1}{D}\partial_{t}\delta\sBulkCo
  &= \partial_{x}^{2}\delta\sBulkCo+\partial_{y}^{2}\delta\sBulkCo+\partial_{z}^{2}\delta\sBulkCo, \nonumber \\
\kappa\partial_{t}\delta\sBulkCo\!\left(t,x,y,0\right)
  &= -\sSurfCo_{0}\partial_{x}\!\left(\partial_{x}\vPotential\!\left(t,x,y,0\right)-\partial_{z}\vStream_{y}\!\left(t,x,y,0\right)+\partial_{y}\vStream_{z}\!\left(t,x,y,0\right)\right) \nonumber \\
  &\phantom{=}\,\,-\sSurfCo_{0}\partial_{y}\!\left(\partial_{y}\vPotential\!\left(t,x,y,0\right)+\partial_{z}\vStream_{x}\!\left(t,x,y,0\right)-\partial_{x}\vStream_{z}\!\left(t,x,y,0\right)\right) \nonumber \\
  &\phantom{=}\,\,-D\partial_{z}\delta\sBulkCo\!\left(t,x,y,0\right), \nonumber \\
0
  &= \sDensity\kinematicViscousity\!\left(2\partial_{x}\partial_{z}\vPotential\!\left(t,x,y,0\right)+\left(\partial_{x}^{2}-\partial_{z}^{2}\right)\vStream_{y}\!\left(t,x,y,0\right)-\partial_{x}\partial_{y}\vStream_{x}\!\left(t,x,y,0\right)+\partial_{z}\partial_{y}\vStream_{z}\!\left(t,x,y,0\right)\right) \nonumber \\
  &\phantom{=}\,\,+k_{e}\kappa\sSurfCo_{0}^{-1}\partial_{x}\delta\sBulkCo\!\left(t,x,y,0\right)+k_{v}\kappa\sSurfCo_{0}^{-1}\partial_{x}\partial_{t}\delta\sBulkCo\!\left(t,x,y,0\right), \nonumber \\
0
  &= \sDensity\kinematicViscousity\!\left(2\partial_{y}\partial_{z}\vPotential\!\left(t,x,y,0\right)-\left(\partial_{y}^{2}-\partial_{z}^{2}\right)\vStream_{x}\!\left(t,x,y,0\right)+\partial_{x}\partial_{y}\vStream_{y}\!\left(t,x,y,0\right)-\partial_{x}\partial_{z}\vStream_{z}\!\left(t,x,y,0\right)\right) \nonumber \\
  &\phantom{=}\,\,+k_{e}\kappa\sSurfCo_{0}^{-1}\partial_{y}\delta\sBulkCo\!\left(t,x,y,0\right)+k_{v}\kappa\sSurfCo_{0}^{-1}\partial_{y}\partial_{t}\delta\sBulkCo\!\left(t,x,y,0\right), \nonumber \\
0
  &= \sDensity\partial_{t}\vPotential\!\left(t,x,y,0\right)+2\sDensity\kinematicViscousity\!\left(\partial_{z}^{2}\vPotential\!\left(t,x,y,0\right)+\partial_{x}\partial_{z}\vStream_{y}\!\left(t,x,y,0\right)-\partial_{y}\partial_{z}\vStream_{x}\!\left(t,x,y,0\right)\right) \nonumber \\
  &\phantom{=}\,\,+\sDensity g\int_{-\infty}^{t}\!\left(\partial_{z}\vPotential\!\left(t,x,y,0\right)+\partial_{x}\vStream_{y}\!\left(t,x,y,0\right)-\partial_{y}\vStream_{x}\!\left(t,x,y,0\right)\right)dt \nonumber \\
  &\phantom{=}\,\,-\sTension_{0}\partial_{x}^{2}\int_{-\infty}^{t}\!\left(\partial_{z}\vPotential\!\left(t,x,y,0\right)+\partial_{x}\vStream_{y}\!\left(t,x,y,0\right)-\partial_{y}\vStream_{x}\!\left(t,x,y,0\right)\right)dt \nonumber \\
  &\phantom{=}\,\,-\sTension_{0}\partial_{y}^{2}\int_{-\infty}^{t}\!\left(\partial_{z}\vPotential\!\left(t,x,y,0\right)+\partial_{x}\vStream_{y}\!\left(t,x,y,0\right)-\partial_{y}\vStream_{x}\!\left(t,x,y,0\right)\right)dt \nonumber
\end{align}
where I already combined equations \eqref{eq:fluid-stress} and \eqref{eq:shear-at-boundary}, \eqref{eq:pressure-at-boundary} and plugged the pressure expression as defined in eq.~\eqref{eq:navier-stokes} directly into the vertical force balance equation.

The assumptions made are: incompressible lower fluid, still upper fluid (air), no chemical or electrical activity, frequency higher than diffusion rate, unsaturated surface film, deep and still bottom, taking everything to linear order (small slopes), and finally, Gibbs elasticity.

\subsection{The boundary conditions}

The boundary conditions for a rigid, weakly vibrating box are
\begin{equation*}
v\!\left(x=\pm\tfrac{\aLength}{2}\right)=v\!\left(y=\pm\tfrac{\aWidth}{2}\right)=v\!\left(z=-\aHeight\right)=\baseVel\!\left(t\right).
\end{equation*}
We solve the problem by solving the homogeneous problem:
\begin{equation*}
v\!\left(x=\pm\tfrac{\aLength}{2}\right)=v\!\left(y=\pm\tfrac{\aWidth}{2}\right)=v\!\left(z=-\aHeight\right)=0,
\end{equation*}
and then treating the wall vibrations as a series of kicks.

As for the pollutant, we take a no-pass boundary condition:
\begin{equation*}
0 = \partial_{x}\delta\sBulkCo\!\left(t,x=\pm\tfrac{\aLength}{2},y,z\right)
  = \partial_{y}\delta\sBulkCo\!\left(t,x,y=\pm\tfrac{\aWidth}{2},z\right)
  = \partial_{z}\delta\sBulkCo\!\left(t,x,y,z=-\aHeight\right).
\end{equation*}
The free surface conditions were already accounted for in system~\eqref{eq:modified-Hansens-model}.

We also impose the gauge condition $0=\nabla\cdot\vStream$.

\subsection{The dispersion relation}

We begin by Fourier transforming the DoFs we're interested in, that is $x$ and $y$. Let $k_{x}=\frac{2\pi m}{\aLength}$, $k_{y}=\frac{2\pi n}{\aWidth}$, $\omega$ be the Fourier parameters, $k:=\sqrt{k_{x}^{2}+k_{y}^{2}}$, and we work with the convention that spatial and temporal parts of the exponent have opposite signs. Then
\begin{align*}
\partial_{z}^{2}\vPotential       &= k^{2}\vPotential, \\
\partial_{z}^{2}\vStream          &= \left(k^{2}-i\frac{\omega}{\kinematicViscousity}\right)\vStream, \\
\partial_{z}^{2}\delta\sBulkCo    &= \left(k^{2}-i\frac{\omega}{D}\right)\delta\sBulkCo, \\
\implies \alpha
  &= \sqrt{k^{2}-i\frac{\omega}{\kinematicViscousity}},\quad \Re\left(\alpha\right)>0, \\
\beta
  &= \sqrt{k^{2}-i\frac{\omega}{D}},\quad \Re\left(\beta\right)>0, \\
\vPotential\!\left(\omega,m,n,z\right)
  &= A_{\vPotential}\!\left(\omega,m,n\right)e^{kz}+B_{\vPotential}\!\left(\omega,m,n\right)e^{-k\left(z+\aHeight\right)}, \\
\vStream\!\left(\omega,m,n,z\right)
  &= A_{\vStream}\!\left(\omega,m,n\right)e^{\alpha z}+B_{\vStream}\!\left(\omega,m,n\right)e^{-\alpha\left(z+\aHeight\right)}, \\
\delta\sBulkCo\!\left(\omega,m,n,z\right)
  &= A_{\delta\sBulkCo}\!\left(\omega,m,n\right)e^{\beta z}+B_{\delta\sBulkCo}\!\left(\omega,m,n\right)e^{-\beta\left(z+\aHeight\right)},
\end{align*}
where I took the choice of expanding about the surface, for numerical reasons.

\subsubsection{Separation of the stream function}

We split $\vStream$ to three components, one that's directed along $\hat{n}:=\frac{1}{k}\left(k_{x},k_{y},0\right)$, $\hat{e}=\hat{n}\times\hat{z}=\frac{1}{k}\left(k_{y}\hat{x}-k_{x}\hat{y}\right)$ and one that's perpendicular to it:
\begin{align*}
\vStream_{T}
  &= \left(\vStream\cdot\hat{n}\right)\hat{n} \\
  &= \left(\frac{k_{x}}{k}\vStream_{x}+\frac{k_{y}}{k}\vStream_{y}\right)\hat{n}, \\
\vStream_{P}+\vStream_{Z}
  &= \vStream-\left(\vStream\cdot\hat{n}\right)\hat{n} \\
  &= \vStream_{x}\hat{x}+\vStream_{y}\hat{y}+\vStream_{z}\hat{z}-\left(\frac{k_{x}}{k}\vStream_{x}+\frac{k_{y}}{k}\vStream_{y}\right)\hat{n}.
\end{align*}
As the names suggest, since $\hat{z}$ by itself is orthogonal to $\hat{n}$, $\vStream\times\hat{n}$ splits naturally to
\begin{align*}
\vStream_{P}
  &= \left(\vStream_{x}-\frac{k_{x}}{k}\left(\frac{k_{x}}{k}\vStream_{x}+\frac{k_{y}}{k}\vStream_{y}\right)\right)\hat{x}+\left(\vStream_{y}-\frac{k_{y}}{k}\left(\frac{k_{x}}{k}\vStream_{x}+\frac{k_{y}}{k}\vStream_{y}\right)\right)\hat{y} \\
  &= \left(\frac{k_{y}}{k}\vStream_{x}-\frac{k_{x}}{k}\vStream_{y}\right)\hat{e} \\
  &= -\left(\vStream\times\hat{n}\right)\times\hat{n}, \\
\vStream_{Z} &= \vStream_{z}\hat{z},
\end{align*}
so let us analogously define $\vStream_{t}=\vStream_{T}\cdot\hat{n}$, $\vStream_{p}=\vStream_{P}\cdot\hat{e}$. Next, the Coulomb gauge $\nabla\cdot\vStream=0$ gives
\begin{align*}
\partial_{z}\vStream_{z}
  &= -ik\,\hat{n}\cdot\vStream \\
  &= -ik\,\vStream_{t}.
\end{align*}
Now we look at the rest of the wall BCs. At $z=-\aHeight$ we have
\begin{align*}
0 &= v\!\left(\omega,m,n,-\aHeight\right) \\
\implies
0 &= ik_{x}\vPotential\!\left(\omega,m,n,-\aHeight\right)+ik_{y}\vStream_{z}\!\left(\omega,m,n,-\aHeight\right)-\partial_{z}\vStream_{y}\!\left(\omega,m,n,-\aHeight\right), \\
0 &= ik_{y}\vPotential\!\left(\omega,m,n,-\aHeight\right)-ik_{x}\vStream_{z}\!\left(\omega,m,n,-\aHeight\right)+\partial_{z}\vStream_{x}\!\left(\omega,m,n,-\aHeight\right) \\
\implies
0 &= ik\,\vPotential\!\left(\omega,m,n,-\aHeight\right)+\partial_{z}\vStream_{p}\!\left(\omega,m,n,-\aHeight\right), \\
0 &= ik\,\vStream_{z}\!\left(\omega,m,n,-\aHeight\right)-\partial_{z}\vStream_{t}\!\left(\omega,m,n,-\aHeight\right) \\
\implies
0 &= \left(k^{2}-\partial_{z}^{2}\right)\vStream_{t}\!\left(\omega,m,n,-\aHeight\right), \\
0 &= \partial_{z}\vPotential\!\left(\omega,m,n,-\aHeight\right)+ik_{x}\vStream_{y}\!\left(\omega,m,n,-\aHeight\right)-ik_{y}\vStream_{x}\!\left(\omega,m,n,-\aHeight\right) \\
  &= \partial_{z}\vPotential\!\left(\omega,m,n,-\aHeight\right)-ik\,\vStream_{p}\!\left(\omega,m,n,-\aHeight\right).
\end{align*}

The free surface BCs take the form
\begin{equation*}
0 = \sSurfCo_{0}k\left(-k\vPotential\!\left(\omega,m,n,0\right)+i\partial_{z}\vStream_{p}\!\left(\omega,m,n,0\right)\right)+\left(D\partial_{z}-i\kappa\omega\right)\delta\sBulkCo\!\left(\omega,m,n,0\right),
\end{equation*}
and
\begin{align*}
0
  &= \sDensity\kinematicViscousity\!\left(2ik_{x}\partial_{z}\vPotential\!\left(\omega,m,n,0\right)+\left(-k_{x}^{2}-\partial_{z}^{2}\right)\vStream_{y}\!\left(\omega,m,n,0\right)+k_{x}k_{y}\vStream_{x}\!\left(\omega,m,n,0\right)+ik_{y}\partial_{z}\vStream_{z}\!\left(\omega,m,n,0\right)\right) \\
  &\phantom{=}\,\,+\kappa\left(ik_{e}+k_{v}\omega\right)k_{x}\sSurfCo_{0}^{-1}\delta\sBulkCo\!\left(\omega,m,n,0\right), \\
0
  &= \sDensity\kinematicViscousity\!\left(2ik_{y}\partial_{z}\vPotential\!\left(\omega,m,n,0\right)+\left(k_{y}^{2}+\partial_{z}^{2}\right)\vStream_{x}\!\left(\omega,m,n,0\right)-k_{x}k_{y}\vStream_{y}\!\left(\omega,m,n,0\right)-ik_{x}\partial_{z}\vStream_{z}\!\left(\omega,m,n,0\right)\right) \\
  &\phantom{=}\,\,+\kappa\left(ik_{e}+k_{v}\omega\right)k_{y}\sSurfCo_{0}^{-1}\delta\sBulkCo\!\left(\omega,m,n,0\right) \\
\implies
0 &= ik\partial_{z}\vStream_{z}\!\left(\omega,m,n,0\right)-\partial_{z}^{2}\vStream_{t}\!\left(\omega,m,n,0\right) \\
  &= \left(k^{2}-\partial_{z}^{2}\right)\vStream_{t}\!\left(\omega,m,n,0\right), \\
0
  &= \sDensity\kinematicViscousity\!\left(i2k\partial_{z}\vPotential\!\left(\omega,m,n,0\right)+\left(k^{2}+\partial_{z}^{2}\right)\vStream_{p}\!\left(\omega,m,n,0\right)\right) \\
  &\phantom{=}\,\,+\kappa\left(ik_{e}+k_{v}\omega\right)k\sSurfCo_{0}^{-1}\delta\sBulkCo\!\left(\omega,m,n,0\right),
\end{align*}
and
\begin{align*}
0
  &= \left(2\sDensity\kinematicViscousity\omega\partial_{z}^{2}+i\left(\sDensity g+\sTension_{0}k^{2}\right)\partial_{z}-i\sDensity\omega^{2}\right)\vPotential\!\left(\omega,m,n,0\right) \\
  &\phantom{=}\,\,+k\left(-i2\sDensity\kinematicViscousity\omega\partial_{z}+\left(\sDensity g+\sTension_{0}k^{2}\right)\right)\vStream_{p}\!\left(\omega,m,n,0\right).
\end{align*}

\subsubsection{Plugging in the functional form}

The zero-flux condition at the bottom implies immediately that
\begin{align*}
0 &= \partial_{z}\delta\sBulkCo\!\left(\omega,m,n,z=-\aHeight\right) \\
  &= A_{\delta\sBulkCo}\!\left(\omega,m,n\right)e^{-\beta \aHeight}-B_{\delta\sBulkCo}\!\left(\omega,m,n\right)e^{\beta \aHeight} \\
\implies B_{\delta\sBulkCo}\!\left(\omega,m,n\right)
  &= e^{-2\beta \aHeight}A_{\delta\sBulkCo}\!\left(\omega,m,n\right).
\end{align*}

Next, we immediately see that since we already have the functional form of $\vStream$, $k^{2}-\partial_{z}^{2}$ becomes $k^{2}-\alpha^{2}$ when it acts on $\vStream_{t}$, and so if we define
\begin{equation*}
\vStream_{t}\!\left(\omega,m,n,z\right)=A_{t}\!\left(\omega,m,n\right)e^{\alpha z}+B_{t}\!\left(\omega,m,n\right)e^{-\alpha\left(z+\aHeight\right)}
\end{equation*}
then
\begin{equation*}
0=\vStream_{t}\!\left(\omega,m,n,0\right)=\vStream_{t}\!\left(\omega,m,n,-\aHeight\right)
\end{equation*}
entails
\begin{align*}
0 &= A_{t}\!\left(\omega,m,n\right) = A_{t}\!\left(\omega,m,n\right)e^{-\alpha \aHeight}-B_{t}\!\left(\omega,m,n\right) \\
\implies \vStream_{t} &= \partial_{z}\vStream_{z}=0
\end{align*}
and so $\vStream_{z}=0$ also.

Likewise, let
\begin{equation*}
\vStream_{p}\!\left(\omega,m,n,z\right)=A_{p}\!\left(\omega,m,n\right)e^{\alpha z}+B_{p}\!\left(\omega,m,n\right)e^{-\alpha\left(z+\aHeight\right)}.
\end{equation*}
Then we have five equations left so far. First, two bottom no-pass conditions:
\begin{align*}
0 &= ik^{2}\vPotential\!\left(\omega,m,n,-\aHeight\right)+k\partial_{z}\vStream_{p}\!\left(\omega,m,n,-\aHeight\right), \\
0 &= \partial_{z}\vPotential\!\left(\omega,m,n,-\aHeight\right)-ik\,\vStream_{p}\!\left(\omega,m,n,-\aHeight\right) \\
\implies
0 &= A_{\vPotential}\!\left(\omega,m,n\right)e^{-\aHeight k}+B_{\vPotential}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-i\frac{\alpha}{k}\left(A_{p}\!\left(\omega,m,n\right)e^{-\aHeight\alpha}-B_{p}\!\left(\omega,m,n\right)\right), \\
0 &= A_{\vPotential}\!\left(\omega,m,n\right)e^{-\aHeight k}-B_{\vPotential}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-i\left(A_{p}\!\left(\omega,m,n\right)e^{-\aHeight\alpha}+B_{p}\!\left(\omega,m,n\right)\right) \\
\implies
0 &= 2e^{-\aHeight k}A_{\vPotential}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-i\left(\frac{\alpha}{k}+1\right)e^{-\aHeight\alpha}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+i\left(\frac{\alpha}{k}-1\right)B_{p}\!\left(\omega,m,n\right), \\
0 &= 2B_{\vPotential}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-i\left(\frac{\alpha}{k}-1\right)e^{-\aHeight\alpha}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+i\left(\frac{\alpha}{k}+1\right)B_{p}\!\left(\omega,m,n\right),
\end{align*}
then three free boundary conditions, the tangential and normal stress balances and the concentration flux equation:
\begin{align*}
0
  &= 2\!\left(A_{\vPotential}\!\left(\omega,m,n\right)-B_{\vPotential}\!\left(\omega,m,n\right)e^{-\aHeight k}\right) \\
  &\phantom{=}\,\,-i\!\left(1+\frac{\alpha^{2}}{k^{2}}\right)\!\left(A_{p}\!\left(\omega,m,n\right)+B_{p}\!\left(\omega,m,n\right)e^{-\aHeight\alpha}\right) \\
  &\phantom{=}\,\,+\frac{1}{\sDensity\kinematicViscousity D}\!\left(\frac{k_{e}-ik_{v}\omega}{k}\right)\!\left(1+e^{-2\beta \aHeight}\right)\tilde{A}_{\delta\sBulkCo}\!\left(\omega,m,n\right), \\
0
  &= \left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}-i\left(\frac{\sDensity}{\sTension_{0}k}\left(\frac{\omega}{k}\right)^{2}-\frac{\sDensity g}{\sTension_{0}k^{2}}-1\right)\right)A_{\vPotential}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}-i\left(\frac{\sDensity}{\sTension_{0}k}\left(\frac{\omega}{k}\right)^{2}+\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)e^{-\aHeight k}B_{\vPotential}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(-i\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}\frac{\alpha}{k}+\left(\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(i\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}\frac{\alpha}{k}+\left(\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)e^{-\aHeight\alpha}B_{p}\!\left(\omega,m,n\right), \\
0
  &= -\left(A_{\vPotential}\!\left(\omega,m,n\right)+e^{-\aHeight k}B_{\vPotential}\!\left(\omega,m,n\right)\right) \\
  &\phantom{=}\,\,+i\frac{\alpha}{k}\!\left(A_{p}\!\left(\omega,m,n\right)-e^{-\aHeight\alpha}B_{p}\!\left(\omega,m,n\right)\right) \\
  &\phantom{=}\,\,+\frac{1}{k^{2}}\!\left(\frac{1}{\kappa}\beta\!\left(1-e^{-2\beta \aHeight}\right)-i\frac{1}{D}\omega\!\left(1+e^{-2\beta \aHeight}\right)\right)\tilde{A}_{\delta\sBulkCo}\!\left(\omega,m,n\right),
\end{align*}
where I defined $\tilde{A}_{\delta\sBulkCo}:=D\kappa\sSurfCo_{0}^{-1}A_{\delta\sBulkCo}$, to make the physical units identical to those of $\vPotential$, $\vStream$. Now we plug in the expressions for $A_{\vPotential}\!\left(\omega,m,n\right)$, $B_{\vPotential}\!\left(\omega,m,n\right)$:
\begin{align*}
0
  &= \left(\left(\frac{\alpha^{2}}{k^{2}}+1\right)e^{-\aHeight k}-\left(\frac{\alpha}{k}+1\right)e^{-\aHeight\alpha}+\left(\frac{\alpha}{k}-1\right)e^{-\aHeight\left(\alpha+2k\right)}\right)A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(\left(\frac{\alpha}{k}-1\right)-\left(\frac{\alpha}{k}+1\right)e^{-2\aHeight k}+\left(\frac{\alpha^{2}}{k^{2}}+1\right)e^{-\aHeight\left(\alpha+k\right)}\right)B_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+i\frac{1}{\sDensity\kinematicViscousity D}\!\left(\frac{k_{e}-ik_{v}\omega}{k}\right)\!\left(1+e^{-2\beta \aHeight}\right)e^{-\aHeight k}\tilde{A}_{\delta\sBulkCo}\!\left(\omega,m,n\right), \\
0
  &= \tfrac{1}{2}\!\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}+i\left(-\frac{\sDensity}{\sTension_{0}k}\left(\frac{\omega}{k}\right)^{2}+\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)\!\left(\frac{\alpha}{k}+1\right)e^{-\aHeight\alpha}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\tfrac{1}{2}\!\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}-i\left(\frac{\sDensity}{\sTension_{0}k}\left(\frac{\omega}{k}\right)^{2}+\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)\!\left(\frac{\alpha}{k}-1\right)e^{-\aHeight\left(\alpha+2k\right)}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}\frac{\alpha}{k}+i\frac{\sDensity g}{\sTension_{0}k^{2}}+i\right)e^{-\aHeight k}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\tfrac{1}{2}\!\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}+i\left(-\frac{\sDensity}{\sTension_{0}k}\left(\frac{\omega}{k}\right)^{2}+\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)\!\left(\frac{\alpha}{k}-1\right)B_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\tfrac{1}{2}\!\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}-i\left(\frac{\sDensity}{\sTension_{0}k}\left(\frac{\omega}{k}\right)^{2}+\frac{\sDensity g}{\sTension_{0}k^{2}}+1\right)\right)\!\left(\frac{\alpha}{k}+1\right)e^{-2\aHeight k}B_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(\frac{2\sDensity\kinematicViscousity}{\sTension_{0}}\frac{\omega}{k}\frac{\alpha}{k}-i\frac{\sDensity g}{\sTension_{0}k^{2}}-i\right)e^{-\aHeight\left(\alpha+k\right)}B_{p}\!\left(\omega,m,n\right), \\
0
  &= \left(\left(\frac{\alpha}{k}+1\right)e^{-\aHeight\alpha}+\left(\frac{\alpha}{k}-1\right)e^{-\aHeight\left(\alpha+2k\right)}-2\frac{\alpha}{k}e^{-\aHeight k}\right)A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\left(\left(\frac{\alpha}{k}-1\right)+\left(\frac{\alpha}{k}+1\right)e^{-2\aHeight k}-2\frac{\alpha}{k}e^{-\aHeight\left(\alpha+k\right)}\right)B_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+i2\!\left(\frac{1}{\kappa}\beta\!\left(1-e^{-2\beta \aHeight}\right)-i\frac{1}{D}\omega\!\left(1+e^{-2\beta \aHeight}\right)\right)\frac{1}{k^{2}}e^{-\aHeight k}\tilde{A}_{\delta\sBulkCo}\!\left(\omega,m,n\right).
\end{align*}
Let us define $r=\frac{\alpha}{k}$, $\tilde{E}\!\left(\omega\right):=\frac{k_{e}-ik_{v}\omega}{\sDensity\kinematicViscousity D}$, $\gamma_{d}:=\frac{D\beta}{\kappa}$, $\gamma_{c}=\frac{\sTension_{0}k}{\sDensity\kinematicViscousity}$, $\omega_{cp}^{2}=\frac{\sTension_{0}k^{3}}{\sDensity}$, $\omega_{0}^{2}=gk+\omega_{cp}^{2}$, then the system becomes
\begin{align}
0
  &= \left(\left(r^{2}+1\right)e^{-\aHeight k}-\left(r+1\right)e^{-\aHeight\alpha}+\left(r-1\right)e^{-\aHeight\left(\alpha+2k\right)}\right)A_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,+\left(\left(r-1\right)-\left(r+1\right)e^{-2\aHeight k}+\left(r^{2}+1\right)e^{-\aHeight\left(\alpha+k\right)}\right)B_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,+i\frac{\tilde{E}\!\left(\omega\right)}{k}\!\left(1+e^{-2\beta \aHeight}\right)e^{-\aHeight k}\tilde{A}_{\delta\sBulkCo}\!\left(\omega,m,n\right), \label{eq:model-final-form} \\
0
  &= \tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}+i\frac{-\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r+1\right)e^{-\aHeight\alpha}A_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,+\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}-i\frac{\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r-1\right)e^{-\aHeight\left(\alpha+2k\right)}A_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,-\left(2r\frac{\omega}{\gamma_{c}}+i\frac{\omega_{0}^{2}}{\omega_{cp}^{2}}\right)e^{-\aHeight k}A_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,-\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}+i\frac{-\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r-1\right)B_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,-\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}-i\frac{\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r+1\right)e^{-2\aHeight k}B_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,+\left(2r\frac{\omega}{\gamma_{c}}-i\frac{\omega_{0}^{2}}{\omega_{cp}^{2}}\right)e^{-\aHeight\left(\alpha+k\right)}B_{p}\!\left(\omega,m,n\right), \nonumber \\
0
  &= \left(\left(r+1\right)e^{-\aHeight\alpha}+\left(r-1\right)e^{-\aHeight\left(\alpha+2k\right)}-2re^{-\aHeight k}\right)A_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,-\left(\left(r-1\right)+\left(r+1\right)e^{-2\aHeight k}-2re^{-\aHeight\left(\alpha+k\right)}\right)B_{p}\!\left(\omega,m,n\right) \nonumber \\
  &\phantom{=}\,\,+i2\frac{\gamma_{d}\!\left(1-e^{-2\beta \aHeight}\right)-i\omega\!\left(1+e^{-2\beta \aHeight}\right)}{Dk^{2}}e^{-\aHeight k}\tilde{A}_{\delta\sBulkCo}\!\left(\omega,m,n\right). \nonumber
\end{align}
Next step would be to compute the determinant. Let us define the coefficients thus:
\begin{align*}
x_{111} &= r^{2}+1, \\
x_{112} &= -\left(r+1\right), \\
x_{113} &= r-1, \\
x_{121} &= r-1, \\
x_{122} &= -\left(r+1\right), \\
x_{123} &= r^{2}+1, \\
x_{131} &= i\frac{\tilde{E}\!\left(\omega\right)}{k}\!\left(1+e^{-2\beta \aHeight}\right), \\
x_{211} &= -\left(2r\frac{\omega}{\gamma_{c}}+i\frac{\omega_{0}^{2}}{\omega_{cp}^{2}}\right), \\
x_{212} &= \tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}+i\frac{-\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r+1\right), \\
x_{213} &= \tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}-i\frac{\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r-1\right), \\
x_{221} &= -\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}+i\frac{-\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r-1\right), \\
x_{222} &= -\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}-i\frac{\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r+1\right), \\
x_{223} &= 2r\frac{\omega}{\gamma_{c}}-i\frac{\omega_{0}^{2}}{\omega_{cp}^{2}}, \\
x_{311} &= -2r, \\
x_{312} &= r+1, \\
x_{313} &= r-1, \\
x_{321} &= -\left(r-1\right), \\
x_{322} &= -\left(r+1\right), \\
x_{323} &= 2r, \\
x_{331} &= i2\frac{\gamma_{d}\!\left(1-e^{-2\beta \aHeight}\right)-i\omega\!\left(1+e^{-2\beta \aHeight}\right)}{Dk^{2}},
\end{align*}
so we need to compute the zeros in $\omega$ of the determinant of the matrix
\begin{equation*}
\begin{pmatrix}
x_{111}e^{-\aHeight k}+x_{112}e^{-\aHeight\alpha}+x_{113}e^{-\aHeight\left(\alpha+2k\right)} & x_{121}+x_{122}e^{-2\aHeight k}+x_{123}e^{-\aHeight\left(\alpha+k\right)} & x_{131}e^{-\aHeight k}\\
x_{211}e^{-\aHeight k}+x_{212}e^{-\aHeight\alpha}+x_{213}e^{-\aHeight\left(\alpha+2k\right)} & x_{221}+x_{222}e^{-2\aHeight k}+x_{223}e^{-\aHeight\left(\alpha+k\right)} & 0\\
x_{311}e^{-\aHeight k}+x_{312}e^{-\aHeight\alpha}+x_{313}e^{-\aHeight\left(\alpha+2k\right)} & x_{321}+x_{322}e^{-2\aHeight k}+x_{323}e^{-\aHeight\left(\alpha+k\right)} & x_{331}e^{-\aHeight k}
\end{pmatrix}
\end{equation*}
which is identical to the the same, multiplied by $e^{-2\aHeight k}$, so we end up with
\begin{align*}
p\!\left(\omega\right)
  &= \det\!\begin{pmatrix}
x_{111}+x_{112}e^{-\aHeight\left(\alpha-k\right)}+x_{113}e^{-\aHeight\left(\alpha+k\right)} & x_{121}+x_{122}e^{-2\aHeight k}+x_{123}e^{-\aHeight\left(\alpha+k\right)} & x_{131}\\
x_{211}+x_{212}e^{-\aHeight\left(\alpha-k\right)}+x_{213}e^{-\aHeight\left(\alpha+k\right)} & x_{221}+x_{222}e^{-2\aHeight k}+x_{223}e^{-\aHeight\left(\alpha+k\right)} & 0\\
x_{311}+x_{312}e^{-\aHeight\left(\alpha-k\right)}+x_{313}e^{-\aHeight\left(\alpha+k\right)} & x_{321}+x_{322}e^{-2\aHeight k}+x_{323}e^{-\aHeight\left(\alpha+k\right)} & x_{331}
\end{pmatrix} \\
  &= x_{131}\!\left(x_{211}+x_{212}e^{-\aHeight\left(\alpha-k\right)}+x_{213}e^{-\aHeight\left(\alpha+k\right)}\right)\cdot\left(x_{321}+x_{322}e^{-2\aHeight k}+x_{323}e^{-\aHeight\left(\alpha+k\right)}\right) \\
  &\phantom{=}\,\,-x_{131}\!\left(x_{221}+x_{222}e^{-2\aHeight k}+x_{223}e^{-\aHeight\left(\alpha+k\right)}\right)\cdot\left(x_{311}+x_{312}e^{-\aHeight\left(\alpha-k\right)}+x_{313}e^{-\aHeight\left(\alpha+k\right)}\right) \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{111}+x_{112}e^{-\aHeight\left(\alpha-k\right)}+x_{113}e^{-\aHeight\left(\alpha+k\right)}\right)\cdot\left(x_{221}+x_{222}e^{-2\aHeight k}+x_{223}e^{-\aHeight\left(\alpha+k\right)}\right) \\
  &\phantom{=}\,\,-x_{331}\!\left(x_{121}+x_{122}e^{-2\aHeight k}+x_{123}e^{-\aHeight\left(\alpha+k\right)}\right)\cdot\left(x_{211}+x_{212}e^{-\aHeight\left(\alpha-k\right)}+x_{213}e^{-\aHeight\left(\alpha+k\right)}\right) \\
  &= x_{131}\!\left(x_{211}x_{321}-x_{221}x_{311}\right) \\
  &\phantom{=}\,\,+x_{131}\!\left(x_{211}x_{322}-x_{222}x_{311}\right)e^{-2\aHeight k} \\
  &\phantom{=}\,\,+x_{131}\!\left(x_{212}x_{321}-x_{221}x_{312}\right)e^{-\aHeight\left(\alpha-k\right)} \\
  &\phantom{=}\,\,+x_{131}\!\left(x_{211}x_{323}+x_{212}x_{322}+x_{213}x_{321}-x_{221}x_{313}-x_{222}x_{312}-x_{223}x_{311}\right)e^{-\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,+x_{131}\!\left(x_{213}x_{322}-x_{222}x_{313}\right)e^{-\aHeight\left(\alpha+3k\right)} \\
  &\phantom{=}\,\,+x_{131}\!\left(x_{212}x_{323}-x_{223}x_{312}\right)e^{-2\aHeight\alpha} \\
  &\phantom{=}\,\,+x_{131}\!\left(x_{213}x_{323}-x_{223}x_{313}\right)e^{-2\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{111}x_{221}-x_{121}x_{211}\right) \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{111}x_{222}-x_{122}x_{211}\right)e^{-2\aHeight k} \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{112}x_{221}-x_{121}x_{212}\right)e^{-\aHeight\left(\alpha-k\right)} \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{111}x_{223}+x_{112}x_{222}+x_{113}x_{221}-x_{121}x_{213}-x_{122}x_{212}-x_{123}x_{211}\right)e^{-\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{113}x_{222}-x_{122}x_{213}\right)e^{-\aHeight\left(\alpha+3k\right)} \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{112}x_{223}-x_{123}x_{212}\right)e^{-2\aHeight\alpha} \\
  &\phantom{=}\,\,+x_{331}\!\left(x_{113}x_{223}-x_{123}x_{213}\right)e^{-2\aHeight\left(\alpha+k\right)} \\
  &= \frac{i}{\omega_{cp}^{2}}x_{131}\!\left(r-1\right)\!\left(r\omega^{2}-\left(r-1\right)\omega_{0}^{2}\right) \\
  &\phantom{=}\,\,+\frac{i}{\omega_{cp}^{2}}x_{131}\!\left(r+1\right)\!\left(r\omega^{2}+\left(r+1\right)\omega_{0}^{2}\right)e^{-2\aHeight k} \\
  &\phantom{=}\,\,-\frac{i}{\omega_{cp}^{2}}x_{131}\,8r\omega_{0}^{2}\,e^{-\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,-\frac{i}{\omega_{cp}^{2}}x_{131}\!\left(r+1\right)\!\left(r\omega^{2}-\left(r+1\right)\omega_{0}^{2}\right)e^{-2\aHeight\alpha} \\
  &\phantom{=}\,\,-\frac{i}{\omega_{cp}^{2}}x_{131}\!\left(r-1\right)\!\left(r\omega^{2}+\left(r-1\right)\omega_{0}^{2}\right)e^{-2\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,+ix_{331}\tfrac{1}{2}\!\left(r-1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}-\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r-1\right)^{2}\frac{\omega}{\gamma_{c}}\right) \\
  &\phantom{=}\,\,+ix_{331}\tfrac{1}{2}\!\left(r+1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}+\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r+1\right)^{2}\frac{\omega}{\gamma_{c}}\right)e^{-2\aHeight k} \\
  &\phantom{=}\,\,-ix_{331}4r\!\left(i\left(r^{2}+3\right)\frac{\omega}{\gamma_{c}}+\frac{\omega^{2}}{\omega_{cp}^{2}}\right)e^{-\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,+ix_{331}\tfrac{1}{2}\!\left(r+1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}-\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r+1\right)^{2}\frac{\omega}{\gamma_{c}}\right)e^{-2\aHeight\alpha} \\
  &\phantom{=}\,\,+ix_{331}\tfrac{1}{2}\!\left(r-1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}+\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r-1\right)^{2}\frac{\omega}{\gamma_{c}}\right)e^{-2\aHeight\left(\alpha+k\right)}.
\end{align*}

\subsubsection{No surfactant}

In this case, the diffusion equation vanishes and so does the tension from the stress balance equations at the surface (when the concentration does less than allow a full 2D cover of the surface, the elastic constants go to $0$). Then model~\eqref{eq:model-final-form} takes the reduced form
\begin{align*}
0
  &= \left(\left(r^{2}+1\right)e^{-\aHeight k}-\left(r+1\right)e^{-\aHeight\alpha}+\left(r-1\right)e^{-\aHeight\left(\alpha+2k\right)}\right)A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(\left(r-1\right)-\left(r+1\right)e^{-2\aHeight k}+\left(r^{2}+1\right)e^{-\aHeight\left(\alpha+k\right)}\right)B_{p}\!\left(\omega,m,n\right), \\
0
  &= \tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}+i\frac{-\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r+1\right)e^{-\aHeight\alpha}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}-i\frac{\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r-1\right)e^{-\aHeight\left(\alpha+2k\right)}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\left(2r\frac{\omega}{\gamma_{c}}+i\frac{\omega_{0}^{2}}{\omega_{cp}^{2}}\right)e^{-\aHeight k}A_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}+i\frac{-\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r-1\right)B_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,-\tfrac{1}{2}\!\left(2\frac{\omega}{\gamma_{c}}-i\frac{\omega^{2}+\omega_{0}^{2}}{\omega_{cp}^{2}}\right)\!\left(r+1\right)e^{-2\aHeight k}B_{p}\!\left(\omega,m,n\right) \\
  &\phantom{=}\,\,+\left(2r\frac{\omega}{\gamma_{c}}-i\frac{\omega_{0}^{2}}{\omega_{cp}^{2}}\right)e^{-\aHeight\left(\alpha+k\right)}B_{p}\!\left(\omega,m,n\right),
\end{align*}
and the respective determinant becomes
\begin{align*}
2p\!\left(\omega\right)
  &= \left(r-1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}-\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r-1\right)^{2}\frac{\omega}{\gamma_{c}}\right) \\
  &\phantom{=}\,\,+\left(r+1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}+\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r+1\right)^{2}\frac{\omega}{\gamma_{c}}\right)e^{-2\aHeight k} \\
  &\phantom{=}\,\,-8r\!\left(i\left(r^{2}+3\right)\frac{\omega}{\gamma_{c}}+\frac{\omega^{2}}{\omega_{cp}^{2}}\right)e^{-\aHeight\left(\alpha+k\right)} \\
  &\phantom{=}\,\,+\left(r+1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}-\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r+1\right)^{2}\frac{\omega}{\gamma_{c}}\right)e^{-2\aHeight\alpha} \\
  &\phantom{=}\,\,+\left(r-1\right)\!\left(\frac{\left(r^{2}+1\right)\omega^{2}+\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r-1\right)^{2}\frac{\omega}{\gamma_{c}}\right)e^{-2\aHeight\left(\alpha+k\right)}.
\end{align*}

\subsubsection{Deep container}

In the limit of a deep container (shallow ripples/large $k$), all but the exponent-free terms vanish and we are left with
\begin{align*}
-\frac{1}{r-1}p\!\left(\omega\right)
  &= \frac{1}{\omega_{cp}^{2}}\frac{\tilde{E}\!\left(\omega\right)}{k}\!\left(1+e^{-2\beta \aHeight}\right)\!\left(r\omega^{2}-\left(r-1\right)\omega_{0}^{2}\right) \\
  &\phantom{=}\,\,+\frac{\gamma_{d}\!\left(1-e^{-2\beta \aHeight}\right)-i\omega\!\left(1+e^{-2\beta \aHeight}\right)}{Dk^{2}}\!\left(\frac{\left(r^{2}+1\right)\omega^{2}-\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r-1\right)^{2}\frac{\omega}{\gamma_{c}}\right)
\end{align*}
at polluted cases, and
\begin{equation*}
-\frac{2i}{r-1}p\!\left(\omega\right) = \frac{\left(r^{2}+1\right)\omega^{2}-\left(r^{2}-1\right)\omega_{0}^{2}}{\omega_{cp}^{2}}+i2\left(r-1\right)^{2}\frac{\omega}{\gamma_{c}}
\end{equation*}
in the clean cases.
